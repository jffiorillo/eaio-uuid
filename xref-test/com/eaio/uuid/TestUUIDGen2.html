28      private void timeUUIDThreads(int reps, int noOfthreads) {
29  
30          Thread[] threads = new Thread[noOfthreads];
31          long numberOfUuidsToBeCreates = reps*noOfthreads;
32          AtomicLong counter = new AtomicLong(0);
33          Map<String,Short> uuids = new ConcurrentHashMap<String,Short>();
34  
35          for (int i = 0; i < threads.length; i++)
36          {
37              threads[i] = new Thread(new UUIDRunnable(reps,uuids,counter));
38          }
39  
40          for (Thread t : threads)
41          {
42              t.start();
43          }
44  
45          for (Thread t : threads)
46          {
47              try {
48                  t.join();
49              } catch (InterruptedException e) {
50                  e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
51              }
52          }
53  
54          assertEquals("Should have made " + numberOfUuidsToBeCreates,numberOfUuidsToBeCreates,counter.get());
55          assertEquals("Should have made " + numberOfUuidsToBeCreates + " unique ids",numberOfUuidsToBeCreates,uuids.size());
56  
57      }
58  
59      private class UUIDRunnable implements  Runnable {
60          private final long its;
61          private final Map<String,Short> values;
62          private final AtomicLong counter;
63  
64          public UUIDRunnable(long iterations, Map<String,Short> values, AtomicLong counter) {
65              its = iterations;
66              this.values = values;
67              this.counter = counter;
68          }
69  
70          public void run() {
71              long i = its + 1;
72              while (0 != --i)
73              {
74                  values.put(new UUID().toString(),Short.MIN_VALUE);
75                  counter.incrementAndGet();
76              }
77          }
78      }

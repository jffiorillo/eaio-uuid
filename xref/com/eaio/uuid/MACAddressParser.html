28  package com.eaio.uuid;
29  
30  import java.util.regex.Matcher;
31  import java.util.regex.Pattern;
32  
33  /**
34   * The MAC address parser attempts to find the following patterns:
35   * <ul>
36   * <li>.{1,2}:.{1,2}:.{1,2}:.{1,2}:.{1,2}:.{1,2}</li>
37   * <li>.{1,2}-.{1,2}-.{1,2}-.{1,2}-.{1,2}-.{1,2}</li>
38   * </ul>
39   *
40   * @see <a href="http://johannburkard.de/software/uuid/">UUID</a>
41   * @author <a href="mailto:jb@eaio.com">Johann Burkard</a>
42   * @version $Id: MACAddressParser.java 4714 2012-03-16 11:43:28Z johann $
43   */
44  public class MACAddressParser {
45  	
46  	public static final Pattern MAC_ADDRESS = Pattern.compile("((?:[A-F0-9]{1,2}[:-]){5}[A-F0-9]{1,2})|(?:0x)(\\d{12})(?:.+ETHER)", Pattern.CASE_INSENSITIVE);
47  
48      /**
49       * Attempts to find a pattern in the given String.
50       *
51       * @param in the String, may not be <code>null</code>
52       * @return the substring that matches this pattern or <code>null</code>
53       */
54      static String parse(String in) {
55          Matcher m = MAC_ADDRESS.matcher(in);
56          if (m.find()) {
57              String g = m.group(2);
58              if (g == null) {
59                  g = m.group(1);
60              }
61          	return g == null ? g : g.replace('-', ':');
62          }
63          return null;
64      }
65  
66  }

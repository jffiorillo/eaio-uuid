34  import com.eaio.util.lang.Hex;

36  /**
37   * Creates UUIDs according to the DCE Universal Token Identifier specification.
38   * <p>
39   * All you need to know:
40   * <pre>
41   * UUID u = new UUID();
42   * </pre>
43   *
44   * @see <a href="http://www.opengroup.org/onlinepubs/9629399/apdxa.htm">
45   * http://www.opengroup.org/onlinepubs/9629399/apdxa.htm
46   * </a>
47   * @see <a href="http://www.uddi.org/pubs/draft-leach-uuids-guids-01.txt">
48   * http://www.uddi.org/pubs/draft-leach-uuids-guids-01.txt
49   * </a>
50   * @see <a href="http://johannburkard.de/software/uuid/">UUID</a>
51   * @author <a href="mailto:jb@eaio.de">Johann Burkard</a>
52   * @version $Id: UUID.java 4688 2012-03-09 14:49:49Z johann $
53   */
54  public class UUID implements Comparable<UUID>, Externalizable, Cloneable,
55          IDLEntity {
56  
57      /**
58       * Hasn't ever changed between versions.
59       */
60      static final long serialVersionUID = 7435962790062944603L;
61  
62      /**
63       * The time field of the UUID.
64       *
65       * @serial
66       */
67      public long time;
68  
69      /**
70       * The clock sequence and node field of the UUID.
71       *
72       * @serial
73       */
74      public long clockSeqAndNode;
75  
76      /**
77       * Constructor for UUID. Constructs a new, unique UUID.
78       *
79       * @see UUIDGen#newTime()
80       * @see UUIDGen#getClockSeqAndNode()
81       */
82      public UUID() {
83          this(UUIDGen.newTime(), UUIDGen.getClockSeqAndNode());
84      }
85  
86      /**
87       * Constructor for UUID. Constructs a UUID from two <code>long</code> values.
88       *
89       * @param time the upper 64 bits
90       * @param clockSeqAndNode the lower 64 bits
91       */
92      public UUID(long time, long clockSeqAndNode) {
93          this.time = time;
94          this.clockSeqAndNode = clockSeqAndNode;
95      }
96  
97      /**
98       * Copy constructor for UUID. Values of the given UUID are copied.
99       *
100      * @param u the UUID, may not be <code>null</code>
101      */
102     public UUID(UUID u) {
103         this(u.time, u.clockSeqAndNode);
104     }
105 
106     /**
107      * Parses a textual representation of a UUID.
108      * <p>
109      * No validation is performed. If the {@link CharSequence} is shorter than 36 characters,
110      * {@link ArrayIndexOutOfBoundsException}s will be thrown.
111      *
112      * @param s the {@link CharSequence}, may not be <code>null</code>
113      */
114     public UUID(CharSequence s) {
115         this(Hex.parseLong(s.subSequence(0, 18)), Hex.parseLong(s.subSequence(
116                 19, 36)));
117     }
118 
119     /**
120      * Compares this UUID to another Object. Throws a {@link ClassCastException} if
121      * the other Object is not an instance of the UUID class. Returns a value
122      * smaller than zero if the other UUID is "larger" than this UUID and a value
123      * larger than zero if the other UUID is "smaller" than this UUID.
124      *
125      * @param t the other UUID, may not be <code>null</code>
126      * @return a value &lt; 0, 0 or a value &gt; 0
127      * @see java.lang.Comparable#compareTo(java.lang.Object)
128      * @throws ClassCastException
129      */
130     public int compareTo(UUID t) {
131         if (this == t) {
132             return 0;
133         }
134         if (time > t.time) {
135             return 1;
136         }
137         if (time < t.time) {
138             return -1;
139         }
140         if (clockSeqAndNode > t.clockSeqAndNode) {
141             return 1;
142         }
143         if (clockSeqAndNode < t.clockSeqAndNode) {
144             return -1;
145         }
146         return 0;
147     }
148 
149     /**
150      * Tweaked Serialization routine.
151      */
152     public void writeExternal(ObjectOutput out) throws IOException {
153         out.writeLong(time);
154         out.writeLong(clockSeqAndNode);
155     }
156 
157     /**
158      * Tweaked Serialization routine.
159      */
160     public void readExternal(ObjectInput in) throws IOException {
161         time = in.readLong();
162         clockSeqAndNode = in.readLong();
163     }
164 
165     /**
166      * Returns this UUID as a String.
167      *
168      * @return a String, never <code>null</code>
169      * @see java.lang.Object#toString()
170      * @see #toAppendable(Appendable)
171      */
172     @Override
173     public final String toString() {
174         return toAppendable(null).toString();
175     }
176 
177     /**
178      * Appends a String representation of this to the given {@link StringBuffer} or
179      * creates a new one if none is given.
180      *
181      * @param in the StringBuffer to append to, may be <code>null</code>
182      * @return a StringBuffer, never <code>null</code>
183      * @see #toAppendable(Appendable)
184      */
185     public StringBuffer toStringBuffer(StringBuffer in) {
186         StringBuffer out = in;
187         if (out == null) {
188             out = new StringBuffer(36);
189         }
190         else {
191             out.ensureCapacity(out.length() + 36);
192         }
193         return (StringBuffer) toAppendable(out);
194     }
195 
196     /**
197      * Appends a String representation of this object to the given {@link Appendable} object.
198      * <p>
199      * For reasons I'll probably never understand, Sun has decided to have a number of I/O classes implement
200      * Appendable which forced them to destroy an otherwise nice and simple interface with {@link IOException}s.
201      * <p>
202      * I decided to ignore any possible IOExceptions in this method.
203      *
204      * @param a the Appendable object, may be <code>null</code>
205      * @return an Appendable object, defaults to a {@link StringBuilder} if <code>a</code> is <code>null</code>
206      */
207     public Appendable toAppendable(Appendable a) {
208         Appendable out = a;
209         if (out == null) {
210             out = new StringBuilder(36);
211         }
212         try {
213             Hex.append(out, (int) (time >> 32)).append('-');
214             Hex.append(out, (short) (time >> 16)).append('-');
215             Hex.append(out, (short) time).append('-');
216             Hex.append(out, (short) (clockSeqAndNode >> 48)).append('-');
217             Hex.append(out, clockSeqAndNode, 12);
218         }
219         catch (IOException ex) {
220             // What were they thinking?
221         }
222         return out;
223     }
224 
225     /**
226      * Returns a hash code of this UUID. The hash code is calculated by XOR'ing the
227      * upper 32 bits of the time and clockSeqAndNode fields and the lower 32 bits of
228      * the time and clockSeqAndNode fields.
229      *
230      * @return an <code>int</code> representing the hash code
231      * @see java.lang.Object#hashCode()
232      */
233     @Override
234     public int hashCode() {
235         return (int) ((time >> 32) ^ time ^ (clockSeqAndNode >> 32) ^ clockSeqAndNode);
236     }
237 
238     /**
239      * Clones this UUID.
240      *
241      * @return a new UUID with identical values, never <code>null</code>
242      */
243     @Override
244     public Object clone() {
245         try {
246             return super.clone();
247         }
248         catch (CloneNotSupportedException ex) {
249             // One of Sun's most epic fails.
250             return null;
251         }
252     }
253 
254     /**
255      * Returns the time field of the UUID (upper 64 bits).
256      *
257      * @return the time field
258      */
259     public final long getTime() {
260         return time;
261     }
262 
263     /**
264      * Returns the clock and node field of the UUID (lower 64 bits).
265      *
266      * @return the clockSeqAndNode field
267      */
268     public final long getClockSeqAndNode() {
269         return clockSeqAndNode;
270     }
271 
272     /**
273      * Compares two Objects for equality.
274      *
275      * @see java.lang.Object#equals(Object)
276      * @param obj the Object to compare this UUID with, may be <code>null</code>
277      * @return <code>true</code> if the other Object is equal to this UUID,
278      * <code>false</code> if not
279      */
280     @Override
281     public boolean equals(Object obj) {
282         if (!(obj instanceof UUID)) {
283             return false;
284         }
285         return compareTo((UUID) obj) == 0;
286     }
287 
288     /**
289      * Returns the nil UUID (a UUID whose values are both set to zero).
290      * <p>
291      * Starting with version 2.0, this method does return a new UUID instance every
292      * time it is called. Earlier versions returned one instance. This has now been
293      * changed because this UUID has public, non-final instance fields. Returning a
294      * new instance is therefore more safe.
295      *
296      * @return a nil UUID, never <code>null</code>
297      */
298     public static UUID nilUUID() {
299         return new UUID(0, 0);
300     }
301 
302     /**
303      * Dedicated method which allows UUID object to be created by jaxrs runtime.
304      *
305      * @param uuid String representation of UUID.
306      * @return UUID object.
307      */
308     public static UUID fromString(final String uuid) {
309         return new UUID(uuid);
310     }
